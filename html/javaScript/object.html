class vehicle                           //defining the class
{
   constructor(c, s)                    //defining the constructor method
   {                                    //is the "blueprint" for the objects
      this.color = c;                   //and its properties
      this.speed = s;
   }

const dacia = new vehicle("Red", 50);   //dacia = {color:red, speed:50}
document.write("Color: " + dacia.color
  + ", Speed: " + dacia.speed + "<br>");

dacia.speed = 75;                       //changes the speed property of dacia
document.write("Color: " + dacia.color
   + ", Speed: " + dacia.speed + "<br>");

let renault = new vehicle("Yellow", 65);  //renault = {color:yellow, speed:65}

renault = new vehicle("Blue", 85);        //renault = {color:blue, speed:85}
document.write("Color: " + renault.color  //overwrites the property-values of renault
   + ", Speed: " + renault.speed);

method

Methods are functions which can only be called on object, which are "member" of the same class in which the object was defined. Methods usually change the properties of the objects.
